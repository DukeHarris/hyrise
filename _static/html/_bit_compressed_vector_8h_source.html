76  T get(size_t column, size_t row) const {
77  checkAccess(column, row);
78 
79  T result;
80  register auto offset = _blockOffset(row);
81  register auto colOffset = _offsetForColumn(column);
82  register auto block = _blockPosition(row) + (offset + colOffset) / _bit_width;
83 
84  // Adjust the column offset to the new block
85  colOffset = (offset + colOffset) % _bit_width;
86 
87  uint64_t bounds = _bit_width - colOffset;
88  register uint64_t baseMask = (1ull << _bits[column]) - 1ull;
89 
90  result = (_data[block] >> colOffset) & baseMask;
91 
92  if (bounds < _bits[column]) {
93  colOffset = _bits[column] - bounds;
94  baseMask = (1ull << colOffset) - 1ull;
95  result |= (baseMask & _data[block + 1]) << bounds;
96  }
97  return result;
98  }

100  void set(size_t column, size_t row, T value) {

103  register auto offset = _blockOffset(row);

107  // Adjust the column offset to the new block

110  register uint64_t bounds = _bit_width - colOffset;
111  register uint64_t baseMask = (1ull << _bits[column]) - 1ull;
112  register uint64_t mask = ~(baseMask << colOffset);
113 
114  _data[block] &= mask;
115  _data[block] |= ((uint64_t) value & baseMask) << colOffset;

117  if (bounds < _bits[column]) {
118  mask = ~(baseMask >> bounds);

124  /*
125  Reserve memory for the given number of rows. memory will only be
126  allocated if the number of rows requires a larger number of blocks
127  than before and it can only be incremented.
128  */
129  void reserve(size_t rows) {
130  if (rows > 0 && _blocks(rows) > _blocks(_allocatedBlocks)) {
131  // Allocate the new memory, copy the old memory and swap the
132  // data
133  assert(_blocks(rows) > 0);
134  storage_t *newMemory = _allocate(_blocks(rows));
135  if (_allocatedBlocks > 0)
136  std::memcpy(newMemory, _data, _allocatedBlocks * sizeof(storage_t));
137 
138  std::swap(_data, newMemory);
139 
140  // Only deallocate if there was something allocated
141  if (newMemory != nullptr)
142  Allocator::Strategy::deallocate(newMemory, _allocatedBlocks * sizeof(storage_t));
143 
144  // set new allocarted blocks
145  _allocatedBlocks = _blocks(rows);
146  }
147  }
