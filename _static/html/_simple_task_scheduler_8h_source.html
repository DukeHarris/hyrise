23  * a simple task scheduler using threadspecific run queues;
24  * each thread is pinned to a dedicated core;
25  * tasks not ready for execution are placed in a central wait set
26  */

50  virtual void pushToQueue(std::shared_ptr<Task> task)
51  {
52  // check if task should be scheduled on specific core
53  int core = task->getPreferredCore();
54 
55  // lock queuesMutex to manipulate queues
56  std::lock_guard<std::mutex> lk2(this->_queuesMutex);
57  if (core >= 0 && core < static_cast<int>(this->_queues)) {
58  //potentially assigns task to a queue blocked by long running task
59  this->_taskQueues[core]->push(task);
60  LOG4CXX_DEBUG(this->_logger, "Task " << std::hex << (void *)task.get() << std::dec << " pushed to queue " << core);
61  } else if (core == NO_PREFERRED_CORE || core >= static_cast<int>(this->_queues)) {
62 
63  if (core < NO_PREFERRED_CORE || core >= static_cast<int>(this->_queues))
64  // Tried to assign task to core which is not assigned to scheduler; assigned to other core, log warning
65  LOG4CXX_WARN(this->_logger, "Tried to assign task " << std::hex << (void *)task.get() << std::dec << " to core " << std::to_string(core) << " which is not assigned to scheduler; assigned it to next available core");
66 
67  // simple strategy to avoid blocking of queues; check if queue is blocked - try a couple of times, otherwise schedule on next queue
68  size_t retries = 0;
69  while (this->_taskQueues[this->_nextQueue]->blocked() && retries < 100) {
70  this->_nextQueue = (this->_nextQueue + 1) % this->_queues;
71  ++retries;
72  }
73 
74  this->_taskQueues[this->_nextQueue]->push(task);
75  //std::cout << "Task " << task->vname() << "; hex " << std::hex << &task << std::dec << " pushed to queue " << this->_nextQueue << std::endl;
76  //round robin on cores
77  this->_nextQueue = (this->_nextQueue + 1) % this->_queues;
78  }
79  }

83  //redistribute tasks to other queues
