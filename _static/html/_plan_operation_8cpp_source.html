130 void _PlanOperation::computeDeferredIndexes() {
131  _field_definition = _indexed_field_definition;
132  if (!_named_field_definition.empty()) {
133 
134  if ((_named_field_definition.size() == 1) && (_named_field_definition[0] == "*")) {
135  for (size_t field_index = 0; field_index < widthOfInputs(input.getTables()); ++field_index) {
136  _field_definition.push_back(field_index);
137  }
138  } else {
139  for (size_t i = 0; i < _named_field_definition.size(); ++i) {
140  _field_definition.push_back(findColumn(_named_field_definition[i]));

142  }
143  }
144  assert(_field_definition.size() >= (_indexed_field_definition.size() + _named_field_definition.size()));

183 void _PlanOperation::operator()() noexcept {
184  if (allDependenciesSuccessful()) {
185  try {
186  LOG4CXX_DEBUG(logger, "Virtual operator called " << vname() << "(" << _operatorId << ")");
187  execute();
188  return;
189  } catch (const std::exception &ex) {
190  setErrorMessage(ex.what());
191  } catch (...) {
192  setErrorMessage("Unknown error");
193  }
194  } else { // dependencies were not successful
195  setErrorMessage(getDependencyErrorMessages());
196  }
197  LOG4CXX_ERROR(logger, this << " " << planOperationName() << " failed: " << getErrorMessage());
198  setState(OpFail);
199 }

203 
204  refreshInput();

224  };
