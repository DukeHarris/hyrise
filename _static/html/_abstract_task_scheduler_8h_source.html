115  virtual void schedule(std::shared_ptr<Task> task) {
116  // simple strategy: check if task is ready to run -> then move to next taskqueue
117  // otherwise store in wait list
118 
119  // lock the task - otherwise, a notify might happen prior to the task being added to the wait set
120  task->lockForNotifications();
121  if (task->isReady())
122  pushToQueue(task);
123  else {
124  task->addReadyObserver(this);
125  std::lock_guard<std::mutex> lk(_setMutex);
126  _waitSet.insert(task);
127  LOG4CXX_DEBUG(_logger, "Task " << std::hex << (void *)task.get() << std::dec << " inserted in wait queue");
128  }
129  task->unlockForNotifications();
130  }

140  /* change the number of threads the task scheduler uses for running tasks;
141  *
142  */
143  void resize(const size_t queues) {
144  // set status to RESIZING
145  // lock scheduler
146  _statusMutex.lock();
147  _status = RESIZING;
148  _statusMutex.unlock();
149 
150  if (queues > _queues) {
151  //set _queues to queues after new queues have been created to new tasks to be assigned to new queues
152  // lock _queue mutex as queues are manipulated
153  std::lock_guard<std::mutex> lk(_queuesMutex);
154  if (static_cast<int>(queues) <= getNumberOfCoresOnSystem()) {
155  for (size_t i = _queues; i < queues; ++i) {
156  _taskQueues.push_back(createTaskQueue(i));
157  }
158  _queues = queues;
159  } else {
160  LOG4CXX_WARN(_logger, "number of queues exceeds available cores; set it to max available cores, which equals to " << std::to_string(getNumberOfCoresOnSystem()));
161  resize(getNumberOfCoresOnSystem());
162  }
163  } else if (queues < _queues) {
164  //set _queues to queues before queues have been deleted to avoid new tasks to be assigned to old queues
165  // lock _queue mutex as queues are manipulated
166  size_t queues_old;
167  {
168  std::lock_guard<std::mutex> lk(_queuesMutex);
169  queues_old = _queues;
170  _queues = queues;
171  //adjust nextQueue to assign a task to in case it is larger than queue size;
172  if (!(_nextQueue < queues))
173  _nextQueue = 0;
174  }
175  for (size_t i = queues_old - 1; i > queues - 1; --i) {
176  stopQueueAndRedistributeTasks(_taskQueues[i], queues);
177  {
178  std::lock_guard<std::mutex> lk(_queuesMutex);
179  delete _taskQueues.back();
180  _taskQueues.pop_back();
181  }
182  }
183  }
184  _statusMutex.lock();
185  _status = RUN;
186  _statusMutex.unlock();
187  }
188  /*
189  * notify scheduler that a given task is ready
190  */
191  void notifyReady(std::shared_ptr<Task> task) {
192  // remove task from wait set
193  _setMutex.lock();
194  int tmp = _waitSet.erase(task);
195  _setMutex.unlock();
196 
197  // if task was found in wait set, schedule task to next queue
198  if (tmp == 1) {
199  LOG4CXX_DEBUG(_logger, "Task " << std::hex << (void *)task.get() << std::dec << " ready to run");
200  pushToQueue(task);
201  } else
202  // should never happen, but check to identify potential race conditions
203  LOG4CXX_ERROR(_logger, "Task that notified to be ready to run was not found / found more than once in waitSet! " << std::to_string(tmp));
204  }

207  * waits for all tasks to finish
208  */
209  void wait() {
210  for (unsigned i = 0; i < _taskQueues.size(); ++i) {
211  _taskQueues[i]->join();
212  }
213  }

216  return _queues;
217  }

219  virtual void shutdown() {
220  _statusMutex.lock();
221  _status = TO_STOP;
222  _statusMutex.unlock();
223  for (unsigned i = 0; i < _taskQueues.size(); ++i) {
224  _taskQueues[i]->stopQueue();
225  }
226  _statusMutex.lock();
227  _status = STOPPED;
228  _statusMutex.unlock();
229  }
