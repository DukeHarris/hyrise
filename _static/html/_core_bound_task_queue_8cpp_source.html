17 void CoreBoundTaskQueue::executeTask() {
18 
19  //infinite thread loop
20  while (1) {
21  //block protected by _threadStatusMutex
22  {
23  std::lock_guard<std::mutex> lk1(_threadStatusMutex);
24  if (_status == TO_STOP)
25  break;
26  }
27  // lock queue to get task
28  std::unique_lock<std::mutex> ul(_queueMutex);
29  // get task and execute
30  if (_runQueue.size() > 0) {
31  std::shared_ptr<Task> task = _runQueue.front();
32  // get first task
33  _runQueue.pop();
34  ul.unlock();
35  if (task) {
36  // set queue to _blocked as we run task; this is a simple mechanism to avoid that further tasks are pushed to this queue if a long running task is executed; check WSSimpleTaskScheduler for task stealing queue
37  _blocked = true;
38  //LOG4CXX_DEBUG(logger, "Started executing task" << std::hex << &task << std::dec << " on core " << _core);
39  // run task
40  //std::cout << "Executed task " << task->vname() << "; hex " << std::hex << &task << std::dec << " on core " << _core<< std::endl;
41  (*task)();
42  LOG4CXX_DEBUG(logger, "Executed task " << task->vname() << "; hex " << std::hex << &task << std::dec << " on core " << _core);
43  // notify done observers that task is done
44  task->notifyDoneObservers();
45  _blocked = false;
46  }
47  }
48  // no task in runQueue -> sleep and wait for new tasks
49  else {
50  //if queue still empty go to sleep and wait until new tasks have been arrived
51  if (_runQueue.size() < 1) {
52  // if thread is about to stop, break execution loop
53  {
54  std::lock_guard<std::mutex> lk1(_threadStatusMutex);
55  if (_status != RUN)
56  continue;
57  }
58  //std::cout << "queue " << _core << " sleeping " << std::endl;
59  _condition.wait(ul);
60  }
61  }
62  }
63 }
